% --- Rules: Relationships ---

% Father and Mother
father(X, Y) :- parent(X, Y), male(X).
mother(X, Y) :- parent(X, Y), female(X).

% Grandparents
grandparent(X, Z) :- parent(X, Y), parent(Y, Z).
grandfather(X, Z) :- grandparent(X, Z), male(X).
grandmother(X, Z) :- grandparent(X, Z), female(X).

% Siblings
% X and Y are siblings if they share the same parent and are not the same person.
sibling(X, Y) :- parent(P, X), parent(P, Y), X \= Y.

% Uncle and Aunt
% X is Y's uncle if X is a sibling of Y's parent and is male.
uncle(X, Y) :- male(X), sibling(X, P), parent(P, Y).
aunt(X, Y) :- female(X), sibling(X, P), parent(P, Y).

% Cousins
% X and Y are cousins if their parents are siblings.
cousin(X, Y) :- parent(P1, X), parent(P2, Y), sibling(P1, P2).

% Ancestor (Recursive rule)
ancestor(X, Y) :- parent(X, Y).
ancestor(X, Y) :- parent(X, Z), ancestor(Z, Y).
